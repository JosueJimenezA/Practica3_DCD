def mostrar_documentacion_tecnica(modulo):
    """Extrae y muestra docstrings de las funciones del mdulo."""
    funciones = inspect.getmembers(modulo, inspect.isfunction)
    
    md_output = f"### Documentacin Tcnica: {modulo.__name__}\n"
    for nombre, func in funciones:
        # Solo mostramos funciones que no empiecen con _
        if not nombre.startswith('_'):
            doc = inspect.getdoc(func) or "Sin documentacin."
            # Formato bonito para el PDF
            md_output += f"#### `def {nombre}{inspect.signature(func)}`\n"
            md_output += f"> {doc}\n\n---\n"
            
    display(Markdown(md_output))
